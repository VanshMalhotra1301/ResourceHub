// C++ Topological Sort using DFS void dfs(int u, vector<vector<int>>& adj, vector<bool>& visited, stack<int>& st) { visited[u] = true; for (int v : adj[u]) { if (!visited[v]) dfs(v, adj, visited, st); } // Push to stack ONLY AFTER all children are processed (Finish Time) st.push(u); } vector<int> topoSort(int V, vector<vector<int>>& adj) { vector<bool> visited(V, false); stack<int> st; for (int i = 0; i < V; i++) { if (!visited[i]) dfs(i, adj, visited, st); } vector<int> res; while (!st.empty()) { res.push_back(st.top()); st.pop(); } return res; }
// C++ Articulation Points using Tarjan's Concept (low/disc) int timer = 0; void dfs(int u, int p, vector<vector<int>>& adj, vector<bool>& vis, vector<int>& disc, vector<int>& low, vector<bool>& isAP) { vis[u] = true; disc[u] = low[u] = ++timer; int children = 0; for (int v : adj[u]) { if (v == p) continue; // Ignore edge to parent if (vis[v]) { // Back-edge found: update low value low[u] = min(low[u], disc[v]); } else { children++; dfs(v, u, adj, vis, disc, low, isAP); // Propagate low value up the tree low[u] = min(low[u], low[v]); // Articulation Point Condition 1: Non-root if (p != -1 && low[v] >= disc[u]) isAP[u] = true; } } // Articulation Point Condition 2: Root with >1 children if (p == -1 && children > 1) isAP[u] = true; }
// C++ Connected Components void dfs(int u, vector<vector<int>>& adj, vector<bool>& visited, int compID, vector<int>& comp) { visited[u] = true; comp[u] = compID; for (int v : adj[u]) { if (!visited[v]) dfs(v, adj, visited, compID, comp); } } vector<int> findComponents(int V, vector<vector<int>>& adj) { vector<bool> visited(V, false); vector<int> comp(V, 0); int compID = 0; for (int i = 0; i < V; i++) { if (!visited[i]) { compID++; dfs(i, adj, visited, compID, comp); } } return comp; // Returns array mapping node to its Component ID }
