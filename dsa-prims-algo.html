Prim's Algorithm (MST)

Time: O(E log V) Space: O(V + E) Greedy Approach

Prim's algorithm finds a Minimum Spanning Tree (MST) for a weighted undirected graph. It builds the MST one vertex at a time, from an arbitrary starting vertex. At each step, it greedily adds the cheapest possible edge from the tree to another vertex.

Prim's Logic
function PrimMST(Graph, startNode):
visited ← {startNode}
mstEdges ← {}
totalWeight ← 0
while visited.size() < Graph.nodes.size():
// Find all edges crossing from visited to unvisited
candidates ← getCrossingEdges(visited)
// Greedy Choice
minEdge ← edge in candidates with smallest weight
// Add to MST
mstEdges.add(minEdge)
visited.add(minEdge.targetNode)
totalWeight ← totalWeight + minEdge.weight
return mstEdges

// C++ Prim's using Priority Queue void primMST(int V, vector<vector<pair<int, int>>>& adj) { priority_queue<pair<int, int>, vector<pair<int, int>>, greater<>> pq; vector<bool> inMST(V, false); int res = 0; pq.push({0, 0}); // {weight, vertex} while (!pq.empty()) { int wt = pq.top().first; int u = pq.top().second; pq.pop(); if (inMST[u]) continue; inMST[u] = true; res += wt; for (auto edge : adj[u]) { int v = edge.first; int weight = edge.second; if (!inMST[v]) { pq.push({weight, v}); } } } }
# Python Prim's using heapq import heapq def prim_mst(V, adj): in_mst = [False] * V pq = [(0, 0)] # (weight, vertex) res = 0 while pq: wt, u = heapq.heappop(pq) if in_mst[u]: continue in_mst[u] = True res += wt for v, weight in adj[u]: if not in_mst[v]: heapq.heappush(pq, (weight, v)) return res
// Java Prim's using PriorityQueue class Pair implements Comparable<Pair> { int v, wt; Pair(int v, int wt) { this.v = v; this.wt = wt; } public int compareTo(Pair that) { return this.wt - that.wt; } } public static int primMST(int V, ArrayList<ArrayList<Pair>> adj) { boolean[] inMST = new boolean[V]; PriorityQueue<Pair> pq = new PriorityQueue<>(); int res = 0; pq.add(new Pair(0, 0)); while (!pq.isEmpty()) { Pair p = pq.poll(); int u = p.v; if (inMST[u]) continue; inMST[u] = true; res += p.wt; for (Pair edge : adj.get(u)) { if (!inMST[edge.v]) { pq.add(new Pair(edge.v, edge.wt)); } } } return res; }
