Shortest Path: Dijkstra

Time: O((V+E) log V) Space: O(V)

Dijkstra's computes the shortest path from a single source to all other nodes. It repeatedly selects the unvisited node with the smallest known distance, and "relaxes" its outgoing edges to see if a cheaper path to neighbors exists.

Dijkstra's Logic
function Dijkstra(Graph, source):
dist[source] ← 0
for each v in Graph (except source):
dist[v] ← ∞
prev[v] ← null
Q ← PriorityQueue(all nodes in Graph)
while Q is not empty:
u ← Q.extract_min() // Node with min dist
for each neighbor v of u:
alt ← dist[u] + weight(u, v)
if alt < dist[v]: // Relaxation
dist[v] ← alt
prev[v] ← u
Q.decrease_priority(v, alt)
return dist, prev
Implementation
// C++ Dijkstra's using Priority Queue #include <iostream> #include <vector> #include <queue> using namespace std; typedef pair<int, int> pii; // {distance, node} void dijkstra(int src, int V, vector<vector<pii>>& adj) { priority_queue<pii, vector<pii>, greater<pii>> pq; vector<int> dist(V, 1e9); // Init distances as Infinity pq.push({0, src}); dist[src] = 0; while (!pq.empty()) { int u = pq.top().second; int d = pq.top().first; pq.pop(); if (d > dist[u]) continue; // Stale pair for (auto& edge : adj[u]) { int v = edge.first; int weight = edge.second; if (dist[u] + weight < dist[v]) { dist[v] = dist[u] + weight; pq.push({dist[v], v}); } } } }
# Python Dijkstra's using heapq import heapq def dijkstra(graph, src, num_nodes): # graph is an adjacency list: { node: [(neighbor, weight), ...] } dist = {i: float('inf') for i in range(num_nodes)} dist[src] = 0 # Priority queue stores (distance, node) pq = [(0, src)] while pq: curr_dist, u = heapq.heappop(pq) # Nodes can get added multiple times, ignore if we found a shorter path if curr_dist > dist[u]: continue for v, weight in graph[u]: alt = dist[u] + weight if alt < dist[v]: dist[v] = alt heapq.heappush(pq, (alt, v)) return dist
// Java Dijkstra's using PriorityQueue import java.util.*; class Node implements Comparable<Node> { int vertex, dist; Node(int v, int d) { vertex = v; dist = d; } public int compareTo(Node other) { return Integer.compare(this.dist, other.dist); } } void dijkstra(int src, int V, List<List<Node>> adj) { int[] dist = new int[V]; Arrays.fill(dist, Integer.MAX_VALUE); dist[src] = 0; PriorityQueue<Node> pq = new PriorityQueue<>(); pq.add(new Node(src, 0)); while (!pq.isEmpty()) { Node current = pq.poll(); int u = current.vertex; // Ignore stale pairs in the queue if (current.dist > dist[u]) continue; for (Node edge : adj.get(u)) { int v = edge.vertex; int weight = edge.dist; if (dist[u] + weight < dist[v]) { dist[v] = dist[u] + weight; pq.add(new Node(v, dist[v])); } } } }
