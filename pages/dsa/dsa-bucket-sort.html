Bucket Sort

Best/Avg: O(n + k) Worst: O(n²) Space: O(n + k) Distribution Sort

Bucket Sort divides the unsorted array into several groups called buckets. Each bucket is then sorted individually (often using a different algorithm like Insertion Sort), and finally, the sorted buckets are concatenated together to form the final sorted array. It works exceptionally well for uniformly distributed data.

Scatter-Gather Strategy
function bucketSort(arr, k): // k = num buckets
min_val ← min(arr), max_val ← max(arr)
range ← max_val - min_val
buckets ← create array of k empty lists
for each num in arr: ← SCATTER PHASE
idx ← ⌊(num - min_val) / (range + 1) * k⌋
buckets[idx].append(num)
for each bucket in buckets: ← SORT BUCKETS
insertionSort(bucket)
result_idx ← 0
for each bucket in buckets: ← GATHER PHASE
for each num in bucket:
arr[result_idx++] ← num
return arr

// Bucket Sort for Integers void bucketSort(vector<int>& arr) { if (arr.empty()) return; int min_val = *min_element(arr.begin(), arr.end()); int max_val = *max_element(arr.begin(), arr.end()); int range = max_val - min_val; int bucketCount = min((int)arr.size(), 5); // Typically adaptive vector<vector<int>> buckets(bucketCount); // Scatter for (int i = 0; i < arr.size(); i++) { int b_idx = ((arr[i] - min_val) * bucketCount) / (range + 1); buckets[b_idx].push_back(arr[i]); } // Sort inside buckets and Gather int index = 0; for (int i = 0; i < bucketCount; i++) { sort(buckets[i].begin(), buckets[i].end()); for (int j = 0; j < buckets[i].size(); j++) { arr[index++] = buckets[i][j]; } } }

// Bucket Sort for Integers void bucketSort(int[] arr) { if (arr.length <= 0) return; int min = arr[0], max = arr[0]; for (int i = 1; i < arr.length; i++) { if (arr[i] < min) min = arr[i]; if (arr[i] > max) max = arr[i]; } int bucketCount = Math.min(arr.length, 5); List<Integer>[] buckets = new List[bucketCount]; for (int i = 0; i < bucketCount; i++) buckets[i] = new ArrayList<>(); // Scatter for (int num : arr) { int b_idx = (num - min) * bucketCount / (max - min + 1); buckets[b_idx].add(num); } // Sort and Gather int index = 0; for (int i = 0; i < bucketCount; i++) { Collections.sort(buckets[i]); for (int num : buckets[i]) { arr[index++] = num; } } }
