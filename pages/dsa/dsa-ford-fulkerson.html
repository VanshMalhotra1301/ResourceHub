Ford-Fulkerson (Edmonds-Karp)

Time: O(V * E^2) Space: O(V + E)

Finds the maximum flow in a network. It repeatedly finds an augmenting path from Source to Sink in the residual graph using BFS (Edmonds-Karp implementation) and pushes the bottleneck capacity through that path until no more paths exist.

Algorithm Logic
function FordFulkerson(Graph, S, T):
max_flow ← 0
Initialize flow on all edges to 0
// Edmonds-Karp uses BFS to find shortest augmenting path
while BFS(Graph, S, T, parent_map) is True:
bottleneck ← ∞
// 1. Find bottleneck capacity along the path
curr ← T
while curr ≠ S:
prev ← parent_map[curr]
residual_cap ← capacity(prev, curr) - flow(prev, curr)
bottleneck ← min(bottleneck, residual_cap)
curr ← prev
// 2. Augment flow along the path
curr ← T
while curr ≠ S:
prev ← parent_map[curr]
flow(prev, curr) ← flow(prev, curr) + bottleneck // Forward
flow(curr, prev) ← flow(curr, prev) - bottleneck // Backward
curr ← prev
max_flow ← max_flow + bottleneck
return max_flow
Implementation (Edmonds-Karp)
// C++ Edmonds-Karp (BFS based Ford-Fulkerson) #include <iostream> #include <vector> #include <queue> #include <algorithm> using namespace std; bool bfs(int V, vector<vector<int>>& resGraph, int s, int t, vector<int>& parent) { fill(parent.begin(), parent.end(), -1); queue<int> q; q.push(s); parent[s] = -2; // Source marker while (!q.empty()) { int u = q.front(); q.pop(); for (int v = 0; v < V; v++) { if (parent[v] == -1 && resGraph[u][v] > 0) { parent[v] = u; if (v == t) return true; q.push(v); } } } return false; } int edmondsKarp(int V, vector<vector<int>>& graph, int s, int t) { vector<vector<int>> resGraph = graph; // Residual graph vector<int> parent(V); int max_flow = 0; while (bfs(V, resGraph, s, t, parent)) { int path_flow = 1e9; // Find bottleneck capacity for (int v = t; v != s; v = parent[v]) { int u = parent[v]; path_flow = min(path_flow, resGraph[u][v]); } // Update residual capacities for (int v = t; v != s; v = parent[v]) { int u = parent[v]; resGraph[u][v] -= path_flow; // Forward edge resGraph[v][u] += path_flow; // Backward edge } max_flow += path_flow; } return max_flow; }
# Python Edmonds-Karp Algorithm from collections import deque def bfs(resGraph, s, t, parent): visited = [False] * len(resGraph) queue = deque([s]) visited[s] = True while queue: u = queue.popleft() for v, cap in enumerate(resGraph[u]): if not visited[v] and cap > 0: queue.append(v) visited[v] = True parent[v] = u if v == t: return True return False def edmonds_karp(graph, s, t): # Create residual graph resGraph = [row[:] for row in graph] parent = [-1] * len(graph) max_flow = 0 while bfs(resGraph, s, t, parent): path_flow = float('inf') v = t # Find bottleneck while v != s: u = parent[v] path_flow = min(path_flow, resGraph[u][v]) v = parent[v] # Augment flow v = t while v != s: u = parent[v] resGraph[u][v] -= path_flow resGraph[v][u] += path_flow v = parent[v] max_flow += path_flow return max_flow
// Java Edmonds-Karp Algorithm import java.util.*; class MaxFlow { boolean bfs(int[][] resGraph, int s, int t, int[] parent) { int V = resGraph.length; boolean[] visited = new boolean[V]; Queue<Integer> queue = new LinkedList<>(); queue.add(s); visited[s] = true; parent[s] = -1; while (!queue.isEmpty()) { int u = queue.poll(); for (int v = 0; v < V; v++) { if (!visited[v] && resGraph[u][v] > 0) { queue.add(v); parent[v] = u; visited[v] = true; if (v == t) return true; } } } return false; } int edmondsKarp(int[][] graph, int s, int t) { int V = graph.length; int[][] resGraph = new int[V][V]; for (int u = 0; u < V; u++) System.arraycopy(graph[u], 0, resGraph[u], 0, V); int[] parent = new int[V]; int max_flow = 0; while (bfs(resGraph, s, t, parent)) { int path_flow = Integer.MAX_VALUE; for (int v = t; v != s; v = parent[v]) { int u = parent[v]; path_flow = Math.min(path_flow, resGraph[u][v]); } for (int v = t; v != s; v = parent[v]) { int u = parent[v]; resGraph[u][v] -= path_flow; resGraph[v][u] += path_flow; } max_flow += path_flow; } return max_flow; } }
